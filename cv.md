function findEvenIndex(arr){
  for(let i = 0; i < arr.length; i++){
    let sumL = 0
    let sumR = 0
    let newArrL = arr.slice(0, i)
    let newArrR = arr.slice(i + 1)
    if(newArrL.length !== 0){
      for (let j = 0; j < newArrL.length; j++) {
      sumL += newArrL[j];
      }
    }else sumL = 0
    if(newArrR.length !== 0){
      for (let j = 0; j < newArrR.length; j++) {
      sumR += newArrR[j];
      }
    }else sumR = 0
    if(sumL === sumR){
      return i
    }
  }
  return -1
}
